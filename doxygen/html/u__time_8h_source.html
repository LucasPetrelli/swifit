39 /*
40  * Standard "struct tm" equivalent.
41  * =NOTE= Differences:-
42  * - Year offset is from 1970 (standard tm is from 1900).
43  * - Wday Sunday is day "1" (standard tm has it as day "0").
44  * - The "Day" field is what standard tm refers to as "yday".
45  * - There is no "date" or Day-of-Month field.
46  * - There is no "isdst" field.
47  */
48 struct tmElements {
49  uint8_t Second;
50  uint8_t Minute;
51  uint8_t Hour;
52  uint8_t Wday; // Day of week, with Sunday as day 1.
53  uint8_t Day; // This is "yday" - Day of Year.
54  uint8_t Month;
55  uint8_t Year; // Offset from 1970.
56 };
